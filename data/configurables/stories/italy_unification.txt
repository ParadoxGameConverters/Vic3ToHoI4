unification_italy = {
    category=unification

    #Keep fully Vic3 syntax?

            is_capital = yes
            OR = {
                is_homeland = cu:north_italian
                is_homeland = cu:south_italian

        any_state = {
            OR = {
                is_homeland = cu:north_italian

        $CULTURE$_countries_array

    target_adj = italian

    ## $CULTURE$ - the cultures in culture_list
    ## $CULTURE_SELF$ - the country's primary culture as set by conversion

        focus={

            y = 1
            cost = 10
            prerequisite = { focus = $TAG$_$TARGET_ADJ$_ambitions }

                factor = 5
            }

            completion_reward = {
                custom_effect_tooltip = $TAG$_unite_$CULTURE_SELF$.tt
                hidden_effect = {
                    every_country = {
                        limit = {
                            is_in_array = { global.$CULTURE_SELF$_countries_array = THIS }
                        }
                        set_country_flag = $TAG$_$CULTURE_SELF$_region_integration_started
                    }
                }
                set_country_flag = {
                    flag = negotiation_in_progress
                    days = 7
                    value = 1
                }
                for_each_scope_loop = {
                    array = global.$CULTURE_SELF$_countries_array

                        limit = {
                            NOT = { tag = ROOT }

                        $TAG$_unite_$TARGET$_effect = yes

            id = $TAG$_expel_others_from_$CULTURE_SELF$_states

            y = 1
            cost = 10

            relative_position_id = $TAG$_expel_others_from_$CULTURE_SELF$_states
            x = -1
            y = 1
            cost = 10
            prerequisite = { focus = $TAG$_unite_$CULTURE_SELF$ focus = $TAG$_expel_others_from_$CULTURE_SELF$_states }
            available = {
                all_of_scopes = {
                    array = global.$CULTURE_SELF$_states_array

            ai_will_do = {
                factor = 5
            }

            completion_reward = {
                for_each_scope_loop = {
                    array = global.$CULTURE_SELF$_states_array
                    add_core_of = ROOT
                }
            }
        }

        focus={
            id = $TAG$_unite_$CULTURE_OTHER$
            text = unite_$CULTURE_OTHER$
            icon = GFX_goal_generic_construction
            relative_position_id = $TAG$_united_$CULTURE_SELF$_states

                factor = 5

                    value = 1
                }
                else = {
                    for_each_scope_loop = {
                        array = global.$CULTURE_OTHER$_nations_array
                        if = {
                            limit = {
                                NOT = { tag = ROOT }

            }
        }

            id = $TAG$_expel_others_from_$CULTURE_OTHER$_states
            text = expel_others_from_$CULTURE_OTHER$_states

            relative_position_id = $TAG$_united_$CULTURE_SELF$_states

                factor = 5

            relative_position_id = $TAG$_expel_others_from_$CULTURE_OTHER$_states

            y = 1

            prerequisite = { focus = $TAG$_unite_$CULTURE_OTHER$ focus = $TAG$_expel_others_from_$CULTURE_OTHER$_states }

                    array = global.$CULTURE_OTHER$_states_array

                factor = 5

            id = $TAG$_proclaim_$TARGET$

        $TAG$_unite_$TARGET$_effect = {
            if = {
                limit = { is_potential_friend_of_ROOT = yes }
                if = {
                    limit = {
                        ROOT = { has_completed_focus = $TAG$_$TARGET_ADJ$_ambitions }

                    country_event = { id = unification_$TARGET$.10 hours = 12 random_hours = 12 }
                }
            }
            else_if = {
                limit = { is_ally_with = ROOT }
                custom_effect_tooltip = $TAG$_already_allied
            }
            else = {
                every_state = {
                    tooltip = $TAG$_claim_states
                    limit = { has_state_flag = original_owner_@PREV is_owned_by_ROOT_or_ally = no }
                    add_claim_by = ROOT
                }
            }
        }

            desc = unification_$TARGET$.10.d

            is_triggered_only = yes
            
            immediate = {

                    check_puppet_willingness_FROM = yes

                name = unification_$TARGET$.10.a

                }
                ai_chance = {
                    factor = 10

                        check_variable = { Root.puppet_willingness > 0 }

                    }
                    modifier = {
                        check_variable = { Root.puppet_willingness > 10 }
                        factor = 2
                    }
                    modifier = {
                        check_variable = { Root.puppet_willingness > 20 }
                        factor = 2

                        factor = 0
