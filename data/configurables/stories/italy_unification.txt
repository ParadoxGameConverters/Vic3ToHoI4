unification_italy = {
    category=unification

    trigger={
        country_has_primary_culture = cu:north_italian
		any_owned_state = {
			is_capital = yes
			is_homeland = cu:north_italian
		}
		any_state = {
			OR = {
				is_homeland = cu:north_italian
				is_homeland = cu:south_italian
			}
			NOT = { is_owned_by = FROM }
		}
    }

    score=102

    block_category=unification


    #start of tree for an italian unifier
    focus={
        id = $TAG$_italian_ambitions
        icon = GFX_goal_generic_construction
        tree_starter = yes
        y = 0
        cost = 10
        ai_will_do = {
            factor = 5
        }

        completion_reward = {
            add_political_power = 50
            add_war_support = 0.1
        }
    }

    # invite north italians to join us
    focus={
        id = $TAG$_unite_north_italians
        icon = GFX_goal_generic_construction
        relative_position_id = $TAG$_italian_ambitions
        x = -1
        y = 1
        cost = 10
        prerequisite = { focus = $TAG$_italian_ambitions }
        ai_will_do = {
            factor = 5
        }

        completion_reward = {
			custom_effect_tooltip = $TAG$_unite_north_italians.tt
			hidden_effect = {
				every_country = {
					limit = {
						OR = {
							is_in_array = { north_italian_nations_array = original_tag }
						}
					}
					set_country_flag = $TAG$_region_integration_started
				}
			}
			set_country_flag = {
				flag = negotiation_in_progress
				days = 7
				value = 1
			}
			else = {
				for_each_loop = {
					array = north_italian_nations_array
					value = v
					limit = {
						NOT = { value = { TAG = $TAG$ } }
						$TAG$_unite_north_italians_effect = yes
					}
				}
			}
        }
    }

    # expel non-italian nations from northern italy
    focus={
        id = $TAG$_expel_others_from_north
        icon = GFX_goal_generic_construction
        relative_position_id = $TAG$_italian_ambitions
        x = 1
        y = 1
        cost = 10
        prerequisite = { focus = $TAG$_italian_ambitions }
        ai_will_do = {
            factor = 5
        }

        completion_reward = {
            #add event chain
        }
    }

    # the north is united
    focus={
        id = $TAG$_united_north
        icon = GFX_goal_generic_construction
        relative_position_id = $TAG$_expel_others_from_north
        x = -1
        y = 1
        cost = 10
        prerequisite = { focus = $TAG$_unite_north_italians focus = $TAG$_expel_others_from_north }
		available = {
			all_of = {
				array = north_italian_states_array
				value = v
				value = { is_fully_controlled_by = $TAG$ }
			}
		}
        ai_will_do = {
            factor = 5
        }

        completion_reward = {
            for_each_loop = {
				array = north_italian_states_array
				value = v
				value = { add_core_of = $TAG }
			}
        }
    }
	
	# invite south italians to join us
    focus={
        id = $TAG$_unite_south_italians
        icon = GFX_goal_generic_construction
        relative_position_id = $TAG$_united_north
        x = -1
        y = 1
        cost = 10
        prerequisite = { focus = $TAG$_united_north }
        ai_will_do = {
            factor = 5
        }

        completion_reward = {
			custom_effect_tooltip = $TAG$_unite_south_italians.tt
			hidden_effect = {
				every_country = {
					limit = {
						OR = {
							is_in_array = { south_italian_nations_array = original_tag }
						}
					}
					set_country_flag = $TAG$_region_integration_started
				}
			}
			set_country_flag = {
				flag = negotiation_in_progress
				days = 7
				value = 1
			}
			else = {
				for_each_loop = {
					array = south_italian_nations_array
					value = v
					limit = {
						NOT = { value = { TAG = $TAG$ } }
						$TAG$_unite_south_italians_effect = yes
					}
				}
			}
        }
    }

    # expel non-italian nations from southern italy
    focus={
        id = $TAG$_expel_others_from_south
        icon = GFX_goal_generic_construction
        relative_position_id = $TAG$_united_north
        x = 1
        y = 1
        cost = 10
        prerequisite = { focus = $TAG$_united_north }
        ai_will_do = {
            factor = 5
        }

        completion_reward = {
            #add event chain
        }
    }

    # the south is united
    focus={
        id = $TAG$_united_south
        icon = GFX_goal_generic_construction
        relative_position_id = $TAG$_expel_others_from_south
        x = -1
        y = 1
        cost = 10
        prerequisite = { focus = $TAG$_unite_south_italians focus = $TAG$_expel_others_from_south }
		available = {
			all_of = {
				array = south_italian_states_array
				value = v
				value = { is_fully_controlled_by = $TAG$ }
			}
		}
        ai_will_do = {
            factor = 5
        }

        completion_reward = {
            for_each_loop = {
				array = south_italian_states_array
				value = v
				value = { add_core_of = $TAG }
			}
        }
    }
	
	# Italy is united, proclaim self as Italy
    focus={
        id = $TAG$_proclaim_italy
        icon = GFX_goal_generic_construction
        relative_position_id = $TAG$_united_south
        x = 0
        y = 1
        cost = 10
        prerequisite = { focus = $TAG$_united_south }
        ai_will_do = {
            factor = 5
        }

        completion_reward = {
            set_cosmetic_tag = ITA
        }
    }

    decisions_categories={
        # borrow decisions from Rome_Strikes_Back\common\decisions\tesb.txt
    }

    decisions={
        # borrow decisions from Rome_Strikes_Back\common\decisions\tesb.txt
    }
	
	scripted_effects={
		$TAG$_unite_north_italians_effect = {
			if = {
				limit = { is_potential_friend_of_ROOT = yes }
				if = {
					limit = {
						ROOT = { has_completed_focus = $TAG$_italian_ambitions }
					}
					country_event = { id = $TAG$_unification_italy.10 hours = 12 random_hours = 12 }
				}
			}
			else_if = {
				limit = { is_ally_with = ROOT }
				custom_effect_tooltip = $TAG$_already_allied
			}
			else = {
				every_state = {
					tooltip = $TAG$_claim_states
					limit = { has_state_flag = original_owner_@PREV is_owned_by_ROOT_or_ally = no }
					add_claim_by = ROOT
				}
			}
		}
	}

	events={
        country_event = {
			id = $TAG$_unification_italy.10
			title = $TAG$_unification_italy.10.t
			desc = $TAG$_unification_italy.10.d
			picture = GFX_goal_generic_improve_relations
			
			is_triggered_only = yes
			
			immediate = {
				hidden_effect = {
					$TAG$_check_puppet_willingness = yes
				}
			}

			# yes
			option = {
				name = $TAG$_unification_italy.10.a
				RME = {
					effect_tooltip = {
						if = {
							limit = { has_dlc = "Together for Victory" }
							set_autonomy = {
								target = ROOT
								autonomy_state = autonomy_puppet
							}
						}
						else = {
							puppet = ROOT
						}
					}
					hidden_effect = { country_event = { id = $TAG$_unification_italy.11 hours = 12 } }
				}
				ai_chance = {
					factor = 10
					modifier = {
						check_variable = { Root.puppet_willingness > 0 }
						factor = 2
					}
					modifier = {
						check_variable = { Root.puppet_willingness > 10 }
						factor = 2
					}
					modifier = {
						check_variable = { Root.puppet_willingness > 20 }
						factor = 2
					}
					modifier = {
						is_subject = yes
						NOT = { is_subject_of = $TAG$ }
						factor = 0
					}
				}
			}

			# no
			option = {
				name = $TAG$_unification_italy.10.b
				$TAG$ = {
					effect_tooltip = {
						create_wargoal = {
							type = puppet_wargoal_focus
							target = ROOT
						}
					}
					hidden_effect = { country_event = { id = $TAG$_unification_italy.12 hours = 12 } }
				}
				ai_chance = {
					factor = 10
					modifier = {
						check_variable = { Root.puppet_willingness < 0 }
						factor = 2
					}
					modifier = {
						check_variable = { Root.puppet_willingness < 10 }
						factor = 2
					}
					modifier = {
						check_variable = { Root.puppet_willingness < 20 }
						factor = 2
					}
					modifier = {
						check_variable = { Root.puppet_willingness < 40 }
						factor = 2
					}
				}
			}
		}

		# [Country] Swears Allegiance
		country_event = {
			id = $TAG$_unification_italy.11
			title = $TAG$_unification_italy.11.t
			desc = $TAG$_unification_italy.11.d
			picture = GFX_goal_generic_improve_relations
			
			is_triggered_only = yes

			option = {
				name = $TAG$_unification_italy.11.a
				if = {
					limit = { has_dlc = "Together for Victory" }
					set_autonomy = {
						target = FROM
						autonomy_state = autonomy_puppet
					}
				}
				else = {
					puppet = FROM
				}
				ai_chance = {
					factor = 10
				}
			}
		}

		# [Country] Stands Defiant
		country_event = {
			id = $TAG$_unification_italy.12
			title = $TAG$_unification_italy.12.t
			desc = $TAG$_unification_italy.12.d
			picture = GFX_goal_generic_improve_relations
			
			is_triggered_only = yes

			option = {
				name = $TAG$_unification_italy.12.a
				create_wargoal = {
					type = puppet_wargoal_focus
					target = FROM
				}
				ai_chance = {
					factor = 10
				}
			}
		}
    }
}
