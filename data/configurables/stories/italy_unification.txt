unification_italy = {

                is_homeland = cu:north_italian
                is_homeland = cu:south_italian

            OR = {
                is_homeland = cu:north_italian

					FROM = {
						exists = yes

                            any_of_scopes = { array = global.north_italian_states_array is_owned_by = PREV }

				days_remove = 60

					}

					FROM = { country_event = { id = $TAG$_unification_italy.20 hours = 12 } }

					base = 100

						check_variable = { From.integrate_willingness < 0 }
