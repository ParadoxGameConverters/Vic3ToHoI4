country_event = {
    id = unification_italy.10
    title = unification_italy.10.t
    desc = unification_italy.10.d
    picture = GFX_goal_generic_improve_relations

    is_triggered_only = yes

    immediate = {
        hidden_effect = {
            check_annex_willingness_FROM = yes
        }
    }

    # yes
    option = {
        name = unification_italy.10.a
        effect_tooltip = {
            annex_country = {
                target = FROM
                transfer_troops = yes
            }
        }
        FROM = {
            hidden_effect = { country_event = { id = unification_italy.11 hours = 12 } }
        }
        ai_chance = {
            factor = 10
            modifier = {
                check_variable = { Root.annex_willingness > 0 }
                factor = 2
            }
            modifier = {
                check_variable = { Root.annex_willingness > 10 }
                factor = 2
            }
            modifier = {
                check_variable = { Root.annex_willingness > 20 }
                factor = 2
            }
            modifier = {
                is_subject = yes
                NOT = { is_subject_of = FROM }
                factor = 0
            }
        }
    }

    # no
    option = {
        name = unification_italy.10.b
        effect_tooltip = {
            create_wargoal = {
                type = annex_everything
                target = FROM
            }
        }
        FROM = {
            hidden_effect = { country_event = { id = unification_italy.12 hours = 12 } }
        }
        ai_chance = {
            factor = 10
            modifier = {
                check_variable = { Root.annex_willingness < 0 }
                factor = 2
            }
            modifier = {
                check_variable = { Root.annex_willingness < 10 }
                factor = 2
            }
            modifier = {
                check_variable = { Root.annex_willingness < 20 }
                factor = 2
            }
            modifier = {
                check_variable = { Root.annex_willingness < 40 }
                factor = 2
            }
        }
    }
}
